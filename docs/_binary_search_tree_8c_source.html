116struct node *removing(struct node *root, void* data, int (*compare)(void *data_one, void *data_two)) {
117 // searching for the item to be deleted
118 if (root == NULL)
119 return NULL;
120 if (compare(root->data, data) == -1){
121 root->right_child = removing(root->right_child, data, compare);
122 }
123 else if (compare(root->data, data) == 1){
124 root->left_child = removing(root->left_child, data, compare);
125 }
126 else {
127 // No Children
128 if (root->left_child == NULL && root->right_child == NULL) {
129 free(root->data);
130 free(root);
131 return NULL;
132 }
133
134 // One Child
135 else if (root->left_child == NULL || root->right_child == NULL) {
136 struct node *temp;

138 temp = root->right_child;

153 }
154 }
155 return root;
